# now we calculate the 5th root of n//45 to find the y:

def iroot(k, n):
   UB = 1
   while pow(UB, k) < n:
       UB *= 2
   LB = UB / 2
   while UB - LB > 1:
       M = (LB + UB) // 2
       midToK = pow(M, k)
       if midToK < n:
           LB = M
       elif n < midToK:
           UB = M
       else:
           return M
   if pow(UB, k) == n:
       return UB
   else:
       return LB
