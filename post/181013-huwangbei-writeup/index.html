import os
def xor(a,b):
    assert len(a)==len(b)
    c=""
    for i in range(len(a)):
        c+=chr(ord(a[i])^ord(b[i]))
    return c
def f(x,k):
    return xor(xor(x,k),7)
def round(M,K):
    L=M[0:27]
    R=M[27:54]
    new_l=R
    new_r=xor(xor(R,L),K)
    return new_l+new_r
def fez(m,K):
    for i in K:
        m=round(m,i)
    return m

LTSHOP#

使用burp的intruder竞争买大辣条，得到超过五个大辣条。

然后买 2**64/5+1 个辣条王，使得需要付的大辣条多到uint装不下，向上溢出为负数，我们就买得起很多的辣条王了，实现一夜暴富。

然后flag就随便买了。
